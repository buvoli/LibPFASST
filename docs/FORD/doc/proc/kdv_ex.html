    function kdv_ex(t, x,beta,Lx) result(u)
    real(pfdp), intent(in)  :: t
    real(pfdp), intent(in)  :: x
    real(pfdp), intent(in)  :: beta
    real(pfdp), intent(in)  :: Lx        
    real(pfdp)  :: u

    real(pfdp) :: a,s

    a=beta*(x-0.375_pfdp*Lx-t)
    s = 2.0_pfdp/(exp(a)+exp(-a))
    u=s*s
    
  end function kdv_ex
